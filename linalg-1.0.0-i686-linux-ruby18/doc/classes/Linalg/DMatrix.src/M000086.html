/*
 * call-seq: minus!(b)
 *
 * In-place matrix subtraction.
 *
 *   c = a - b
 *
 * is equivalent to
 *
 *   c = a.minus!(b)
 *
 * with
 *
 *   a.object_id == c.object_id
 *
 */
VALUE rb_dmatrix_minus_inplace(VALUE self, VALUE rb)
{
    DMatrix* a ;
    DMatrix* b ;
    get_dmatrix(a, self) ;
    get_dmatrix(b, rb) ;

    if( a->vsize != b->vsize ||
        a->hsize != b->hsize)
    {
        raise_dim_error() ;
    }

    DMatrix_minus(a, b) ;
    return self ;
}
