      def self.join_rows(rows)
         row_hsize = rows[0].hsize
         res = DMatrix.reserve(rows.size, row_hsize)
         rows.each_with_index { |row, i|
            unless row.hsize == row_hsize
               raise DimensionError 
            end
            res.replace_row(i, row)
         }
         res
      end
