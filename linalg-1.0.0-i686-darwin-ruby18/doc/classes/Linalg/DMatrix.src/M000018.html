      def self.join_columns(cols)
         col_vsize = cols[0].vsize
         res = DMatrix.reserve(col_vsize, cols.size)
         cols.each_with_index { |col, j|
            unless col.vsize == col_vsize
               raise DimensionError 
            end
            res.replace_column(j, col)
         }
         res
      end
