/*
 * call-seq: plus!(b)
 *
 * In-place matrix addition.
 *
 *   a.object_id == a.plus!(b).object_id
 *
 */
VALUE rb_dmatrix_plus_inplace(VALUE self, VALUE rb)
{
    DMatrix* a ;
    DMatrix* b ;
    get_dmatrix(a, self) ;
    get_dmatrix(b, rb) ;

    if( a->vsize != b->vsize ||
        a->hsize != b->hsize)
    {
        raise_dim_error() ;
    }

    DMatrix_plus(a, b) ;
    return self ;
}
